CREATE TABLE IF NOT EXISTS customer_vouchers (
    id SERIAL PRIMARY KEY,       -- unique row id
    customer_id INTEGER NOT NULL,
    voucher_id INTEGER NOT NULL,
    book_id INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    CONSTRAINT fk_customer FOREIGN KEY (customer_id)
        REFERENCES customers(id) ON DELETE CASCADE,
    CONSTRAINT fk_voucher FOREIGN KEY (voucher_id)
        REFERENCES voucher_types(id) ON DELETE CASCADE,
    CONSTRAINT fk_book FOREIGN KEY (book_id)
        REFERENCES voucher_books(id) ON DELETE CASCADE,
    CONSTRAINT unique_customer_voucher UNIQUE (customer_id, voucher_id, book_id)
);
